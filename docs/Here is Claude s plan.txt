 class BugLogger:
     BUGS_DIR = Path("logs/bugs")

     def log_bug(
         self,
         bug_id: str,           # z.B. "creative_empty_prompt"
         severity: str,         # "critical" | "high" | "medium" | "low"

 Meilenstein 2 — Resilience: Timeout + Retry

 - Zeile 262: result = await agent.run(task) ersetzen durch:
 timeout = float(os.getenv("DELEGATION_TIMEOUT", "120"))
 try:
     result = await asyncio.wait_for(agent.run(task), timeout=timeout)

     raise TimeoutError(
         f"Agent '{to_agent}' hat nicht innerhalb von {timeout}s geantwortet"

 2.3 — Retry-Logik in delegate() einbauen
 - Vor dem try-Block (Zeile 251): Retry-Schleife hinzufügen
 - Max-Retries via DELEGATION_MAX_RETRIES (Default: 1, kein Retry)
 - Retry nur bei TimeoutError und allgemeinen Exceptions, NICHT bei validierten
 Fehlern (Agent-nicht-registriert, Zirkulär → die werden direkt zurückgegeben)
 max_retries = int(os.getenv("DELEGATION_MAX_RETRIES", "1"))
 last_error = None
 for attempt in range(max_retries):
     try:
         result = await asyncio.wait_for(agent.run(task), timeout=timeout)
         break  # Erfolg
     except (asyncio.TimeoutError, Exception) as e:
         last_error = e
         if attempt < max_retries - 1:
             await asyncio.sleep(2 ** attempt)  # 1s, 2s backoff
 else:
     raise last_error
