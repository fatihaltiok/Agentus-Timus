 @dataclass
 class GeneratedTool:
     name: str
     description: str
     code: str           # vollständiger tool.py-Inhalt
     action_id: str
     status: str         # "pending" | "approved" | "rejected" | "active"
     code_length: int    # für Lean-Verifikation

 class ToolGeneratorEngine:
     MAX_CODE_LENGTH: int = 5000  # Sicherheitslimit (×1: als Int für Lean)

     def generate(self, tool_name: str, description: str, parameters: List[str]) -> GeneratedTool
     def validate_ast(self, code: str) -> Tuple[bool, str]  # AST-Sicherheitscheck
     def request_review(self, tool: GeneratedTool) -> None   # Telegram: Code-Preview + Approve/Reject
     def activate(self, action_id: str) -> bool              # importlib.import_module() nach Genehmigung
     def reject(self, action_id: str) -> bool
     def _write_tool_files(self, tool: GeneratedTool) -> Path
